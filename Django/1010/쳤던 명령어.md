def blog(request):
    # db = Blog.objects.all()
    context = {
        'db': db,
    }
    return render(request, 'blog/blog.html', context)

def post(request, pk):
    # db = Blog.objects.get(pk=pk)
    context = {
        'db': db[pk-1],
    }
    return render(request, 'blog/post.html', context)
